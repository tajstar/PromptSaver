function savePrompt() {
  chrome.storage.local.get(['savedPrompts'], function (result) {
    const promptElements = document.querySelectorAll('[data-id]:not(.hidden) .save-prompt');
    const prompts = Array.from(promptElements).map(promptElement => {
      // Clone the prompt element to avoid modifying the original element
      const clonedPromptElement = promptElement.cloneNode(true);

      // Remove the "Save Prompt" button from the cloned prompt element
      const savePromptButton = clonedPromptElement.querySelector('.save-prompt-button');
      if (savePromptButton) {
        savePromptButton.remove();
      }

      return clonedPromptElement.textContent.trim();
    });

    const savedPrompts = result.savedPrompts || [];
    savedPrompts.push(...prompts);
    chrome.storage.local.set({ savedPrompts: savedPrompts }, function () {
      console.log('Prompts saved:', prompts);
    });
  });
}

  function savePrompt(promptText) {
    if (promptText) {
      // Get the existing prompts from local storage
      chrome.storage.local.get(['savedPrompts'], function (result) {
        let savedPrompts = result.savedPrompts || [];
  
        // Check if there are already 100 prompts saved
        if (savedPrompts.length >= 100) {
          alert('Maximum number of prompts (100) reached. Please remove some prompts before adding new ones.');
          return;
        }
  
        // Save the new prompt to the local storage
        savedPrompts.push(promptText);
        chrome.storage.local.set({ 'savedPrompts': savedPrompts }, function () {
          alert('Prompt saved successfully!');
        });
      });
    } else {
      alert('No prompt found to save.');
    }
  }
